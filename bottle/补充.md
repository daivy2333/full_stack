// js/modules/uiController.js

export class UIController {
    constructor() {
        // 缓存 DOM 元素
        this.bottleWrapper = document.getElementById('bottle-display');
        this.visualBottle = document.getElementById('visual-bottle'); // 新增
        this.bottleContentView = document.getElementById('bottle-content-view'); // 新增
        this.bottleMessage = document.getElementById('bottle-message');
        // ... 其他元素缓存
        
        this.initEventListeners();
    }

    initEventListeners() {
        // 绑定点击“视觉瓶子”的事件，点击后打开信件
        if (this.visualBottle) {
            this.visualBottle.addEventListener('click', () => {
                this.openBottle();
            });
        }
        // ... 其他事件绑定
    }

    // === 核心修改：显示捡到的瓶子（此时是未打开状态） ===
    showPickedBottle(data) {
        // 1. 填充数据到隐藏的内容视图中
        this.renderBottleContent(data);
        
        // 2. 重置视图状态
        this.bottleWrapper.classList.remove('hidden');
        this.visualBottle.classList.remove('hidden'); // 显示瓶子图标
        this.bottleContentView.classList.add('hidden'); // 隐藏信件内容
        
        // 3. 添加浮出水面的动画
        this.visualBottle.classList.remove('anim-float-up');
        void this.visualBottle.offsetWidth; // 触发重绘，重置动画
        this.visualBottle.classList.add('anim-float-up');
    }

    // === 新增方法：打开瓶子 ===
    openBottle() {
        // 1. 隐藏瓶子图标
        this.visualBottle.classList.add('hidden');
        
        // 2. 显示信件内容
        this.bottleContentView.classList.remove('hidden');
        
        // 3. 添加内容淡入动画
        this.bottleContentView.classList.add('anim-fade-in');
    }

    renderBottleContent(data) {
        // 这里放原来的数据填充逻辑
        this.bottleMessage.textContent = data.content;
        document.getElementById('bottle-author').textContent = `作者: ${data.author || '匿名'}`;
        // ... 其他数据填充
    }
    
    // ... 其他方法
}
